# Verifier Strategy

## Example 1

Given:

```
package com.qux.foo {
    // anonymous block
    {
        var o:B.<Number> = new B.<Number>;
            // ^ this should fail because Number does not implement L
        o.x;
        o.f(10);
    }

    public class B.<T> extends A.<T> {
        public function B() {
            super();
            trace(this.x.toString(16));
        }
    } // B

    public class A.<T>
        where T is L // T implements L
    {
        public var x = 10;

        public function f.<O>(v:O):void
            where O is L // O implements L
        {
        } // f
    } // A

    public interface L {
    } // L
}
```
